function (x, strip = TRUE) {
	stopifnot(is.list(x))
	require(stringdist)
	r <- vector("list", length = length(x))
	for (i in seq_along(x)) {
		#	i = 1
		if (strip) {
			 ii <- stringdist(stripSensuLato(names(x[i])), unlist(x[i]))			
		}
		else {
			ii <- stringdist(names(x[i]), unlist(x[i]))
		}
		if (any(ii > 0)) {
			r[[i]] <- rank(ii)	
		}
		else {
			r[[i]] <- ii
		}
	}
	names(r) <- names(x)
	
	return(r)
}
